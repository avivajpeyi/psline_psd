import numpy as np
import pandas as pd

DATA_URL = "https://gist.githubusercontent.com/avivajpeyi/67c3129ac75e6e9da1f75209de7ec5fe/raw/849ee5a2bee139ac0cdcce10720630d7a668a92d/lisa_wd_strain"

data = np.loadtxt(DATA_URL, delimiter=" ")
# only keep first 5000 points for speed
data = data[:5000]

data = pd.DataFrame(data, columns=["strain"])
dt = 5.00000079
data["time"] = np.arange(0, len(data) * dt, dt)
data.plot(x="time", y="strain");

from pspline_psd.sample.gibbs_pspline_simple import gibbs_pspline_simple

mcmc = gibbs_pspline_simple(
    data=data.strain,
    Ntotal=200,
    burnin=100,
    degree=3,
    eqSpacedKnots=True,
    compute_psds=True,
)

from pspline_psd.plotting.plot_psd import plot_psd

fig = plot_psd(data, mcmc.psd_quantiles)
